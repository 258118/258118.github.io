import numpy as np

import matplotlib.pyplot as plt

x = np.linspace(0, 10, 100)

y1 = np.sin(x)

y2 = np.cos(x)

plt.plot(x, y1, 'r-', label='sin(x)')

plt.plot(x, y2, 'b--', label='cos(x)')

plt.xlabel("Time (s)")

plt.ylabel("Amplitude")

plt.title("Sine and Cosine Waves")

plt.legend()

plt.grid(True)

plt.show()
